218  static matrix<float> calculate_transformation_matrix(std::array<float, 3> red, std::array<float, 3> green, std::array<float, 3> blue, std::array<float, 3> white_XYZ)
219  {
220  std::vector<float>cc_values{ red[0], green[0], blue[0], red[1], green[1], blue[1], red[2], green[2], blue[2] };
221  matrix<float> cc(3, cc_values);
222 
223  auto cc_inv = cc.invert();
224  auto tmp_vec = cc_inv * std::vector<float>(white_XYZ.begin(), white_XYZ.end());
225 
226  std::vector<float>tmp_mat_data{ tmp_vec[0], 0, 0, 0, tmp_vec[1], 0, 0, 0, tmp_vec[2] };
227  matrix<float> tmp_mat(3, tmp_mat_data);
228 
229  return cc * tmp_mat;
230  }
